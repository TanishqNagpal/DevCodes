Integrating a Payment Gateway using Razorpay

How many times have you went to your favourite e-commmerce or food ordering website and ordered using payment options like UPI, Debit Card, Credit Card, etc ? With this tutorial you can implement the same in your own web-app.

Language/Framework used: Javascript, NodeJS

Last Updated: 15-03-2022

Link to entire code

https://github.com/TanishqNagpal/DevCodes

Step -1

Create a folder and open it in vs-code. In the terminal run the following commands to download the necessary packages. 

            

              npm init
              npm i express
              npm i razorpay
              npm i nodemon
              npm i ejs

Step -2

In the folder, create a file index.js and paste the following startup code to set up the packages and run the server. 

            

const express=require("express");
const Razorpay=require("razorpay");

let app=express(); 

app.listen(3000,function()
{
    
  console.log("server on port 3000");

});

Step -3

Add the following startup code for ejs in index.js file. 

          
            app.set('views','views');
            app.set('view engine', 'ejs');
            app.use(express.urlencoded({extended:false}));

Step -5

Copy the following code in index.js to render the html file and then run the server to see the changes.

              res.render("razorpay.ejs");

            });

Step -6

Now, go to the official razorpay website and generate API keys in test mode. Save both the key_id and key_secret generated.

Step -7

Copy the following code in index.js to initialise razorpay. Please note you have to replace the key_id and key_secret with your own keys for the payment to reflect in your account. 

          
            const razorpay=new Razorpay({

              key_id: "rzp_test_uOlQOrpdrVORzm",
              key_secret:"bqLHAV7Ih5mia212Hnx4h7Qi",
            });

Step -8

Now Create an order from the server for which the user will pay. It will contain the amount to be paid and the currency. A unique order is created for every payment and the order_id generated has to be passed to checkout. Also note the amount specified here is in paise. Copy the following code in index.js 

          
            app.post("/order", function(req,res){

              var options = {
                amount: 50000 ,
                currency: "INR" ,
            
              };
            
              razorpay.orders.create(options, function(err, order){
                
                console.log(order);
                res.json(order);  //to send the json file back to frontend
              });
            });

Step -9

Import axios and razorpay checkout js file using the following code. Copy it in razorpay.ejs file inside the head tag. 

          
            <script src="https://cdn.jsdelivr.net/npm/axios/dist/axios.min.js"></script>
            <script src="https://checkout.razorpay.com/v1/checkout.js"></script>

Step -11

When a post request is made with the callback URL, we get payment_id, order_id and signature in return if the payment is successful.
If the payment fails, an empty object is returned.
Last step is to verify the signature.
Here we are verifying if the payment is captured then it will show the message of "successful Payment". 

          
            app.post("/is-order-complete",function(req,res){

              razorpay.payments.fetch(req.body.razorpay_payment_id).then((paymentDocument)=>{
            
                if(paymentDocument.status =="captured")
                {
                  res.send("Payment successful");
                }
                else{
            
                  res.redirect("/");
                }
            });
          });

Step -12

With this our payment gateway is integrated to our web-app. For testing keys like card number and UPI id for a successful payment, you can find these in the offical razorpay website.
After a payment is done, you can check the details of the transaction in Dashboard -> Transactions on the razorpay site. If the status shows "captured" that means transaction is successful.
The final index.js and razorpay.ejs files are shown below: 

          
const express=require("express");
const Razorpay=require("razorpay");

let app=express(); 
const razorpay=new Razorpay({

  key_id: "rzp_test_uOlQOrpdrVORzm",
  key_secret:"bqLHAV7Ih5mia212Hnx4h7Qi",
});


app.set('views','views');
app.set('view engine', 'ejs');
app.use(express.urlencoded({extended:false}));

app.get("/",function(req,res){

  res.render("razorpay.ejs");
});

app.post("/order", function(req,res){

  var options = {
    amount: 50000 ,
    currency: "INR" ,

  };

  razorpay.orders.create(options, function(err, order){
    
    console.log(order);
    res.json(order);
  });
});

app.post("/is-order-complete",function(req,res){

  razorpay.payments.fetch(req.body.razorpay_payment_id).then((paymentDocument)=>{

    if(paymentDocument.status =="captured")
    {
      res.send("Payment successful");
    }
    else{

      res.redirect("/");
    }
});


});
app.listen(3000,function()
{
  console.log("server on port 3000");
});
